// -------- OWL (Orchestrator) - wise owl with round glasses --------
function createStylizedOwl(color) {
  const g = new THREE.Group()
  const c = new THREE.Color(color)
  const bodyMat = new THREE.MeshStandardMaterial({
    color: c, roughness: 0.25, metalness: 0.35,
    emissive: c, emissiveIntensity: 0.06,
  })
  const accentMat = new THREE.MeshStandardMaterial({ color: 0xe8d5b7, roughness: 0.5 })
  const beakMat = new THREE.MeshStandardMaterial({ color: 0xff8c42, roughness: 0.3, metalness: 0.1 })
  // Body - round plump body
  const body = new THREE.Mesh(new THREE.SphereGeometry(0.12, 12, 12), bodyMat)
  body.scale.set(1, 1.15, 0.9); body.position.y = 0.06; g.add(body)
  // Belly patch
  const belly = new THREE.Mesh(new THREE.SphereGeometry(0.065, 10, 10), accentMat)
  belly.scale.set(1, 1.1, 0.5); belly.position.set(0, 0.04, -0.08); g.add(belly)
  // Head - big round head
  const head = new THREE.Mesh(new THREE.SphereGeometry(0.09, 12, 12), bodyMat)
  head.scale.set(1, 0.9, 0.85); head.position.set(0, 0.18, -0.01); g.add(head)
  // Ear tufts - little triangles
  for (let s = -1; s <= 1; s += 2) {
    const t = new THREE.Mesh(new THREE.ConeGeometry(0.018, 0.045, 4), bodyMat)
    t.position.set(s * 0.04, 0.25, -0.01); t.rotation.z = s * 0.15; g.add(t)
  }
  // Big cute eyes with round glasses frames
  for (let s = -1; s <= 1; s += 2) {
    const eye = new THREE.Mesh(new THREE.SphereGeometry(0.04, 12, 12), new THREE.MeshStandardMaterial({ color: 0xffffff, emissive: 0xffffff, emissiveIntensity: 0.2 }))
    eye.scale.set(1, 0.9, 0.5); eye.position.set(s * 0.05, 0.20, -0.08); g.add(eye)
    const pupil = new THREE.Mesh(new THREE.SphereGeometry(0.028, 10, 10), new THREE.MeshStandardMaterial({ color: 0x1a1a2e }))
    pupil.position.set(s * 0.05, 0.20, -0.06); g.add(pupil)
    const hl = new THREE.Mesh(new THREE.SphereGeometry(0.012, 6, 6), new THREE.MeshStandardMaterial({ color: 0xffffff }))
    hl.position.set(s * 0.055, 0.215, -0.065); g.add(hl)
    const hl2 = new THREE.Mesh(new THREE.SphereGeometry(0.005, 6, 6), new THREE.MeshStandardMaterial({ color: 0xffffff }))
    hl2.position.set(s * 0.043, 0.195, -0.07); g.add(hl2)
    // Glasses frame - round ring
    const frame = new THREE.Mesh(new THREE.TorusGeometry(0.045, 0.008, 8, 16), new THREE.MeshStandardMaterial({ color: 0xffd700, metalness: 0.5, roughness: 0.2 }))
    frame.position.set(s * 0.05, 0.20, -0.055); frame.scale.set(1, 0.9, 0.3); g.add(frame)
  }
  // Glasses bridge
  const bridge = new THREE.Mesh(new THREE.BoxGeometry(0.03, 0.006, 0.006), new THREE.MeshStandardMaterial({ color: 0xffd700, metalness: 0.5 }))
  bridge.position.set(0, 0.20, -0.055); g.add(bridge)
  // Beak - tiny cute triangle
  const beak = new THREE.Mesh(new THREE.ConeGeometry(0.02, 0.03, 4), beakMat)
  beak.rotation.x = 0.3; beak.position.set(0, 0.165, -0.12); g.add(beak)
  // Wings - small rounded
  for (let s = -1; s <= 1; s += 2) {
    const wing = new THREE.Mesh(new THREE.SphereGeometry(0.04, 8, 8), bodyMat)
    wing.scale.set(0.35, 0.7, 0.5); wing.position.set(s * 0.13, 0.07, 0); g.add(wing)
  }
  // Book under wing (orchestrator accessory)
  const bookCovMat = new THREE.MeshStandardMaterial({ color: 0x8b4513, roughness: 0.8 })
  const bookPagMat = new THREE.MeshStandardMaterial({ color: 0xf5f0e0, roughness: 0.5 })
  const book = new THREE.Mesh(new THREE.BoxGeometry(0.06, 0.025, 0.04), bookCovMat)
  book.position.set(-0.1, 0.03, 0.02); g.add(book)
  const bookPage = new THREE.Mesh(new THREE.BoxGeometry(0.052, 0.003, 0.035), bookPagMat)
  bookPage.position.set(-0.1, 0.035, 0.02); g.add(bookPage)
  g.children.forEach(function(ch) { if (ch.isMesh) { ch.userData.isCore = true; ch.userData.isAnimal = true } })
  g.userData.animalType = 'owl'
  g.userData.book = book
  return g
}

  return g

// -------- DOLPHIN (Weather) - dolphin with tiny umbrella --------
function createStylizedDolphin(color) {
  const g = new THREE.Group()
  const c = new THREE.Color(color)
  const mat = new THREE.MeshStandardMaterial({ color: c, roughness: 0.1, metalness: 0.5, emissive: c, emissiveIntensity: 0.04 })
  const bellyMat = new THREE.MeshStandardMaterial({ color: 0xccf0ff, roughness: 0.2, metalness: 0.1 })
  const umbrellaMat = new THREE.MeshStandardMaterial({ color: 0x4ade80, roughness: 0.5 })
  const umbrellaAccent = new THREE.MeshStandardMaterial({ color: 0x22c55e, roughness: 0.4 })
  // Body - curved sleek
  const body = new THREE.Mesh(new THREE.CapsuleGeometry(0.05, 0.18, 8, 12), mat)
  body.rotation.x = Math.PI * 0.06; body.position.y = 0.03; g.add(body)
  // Belly
  const belly = new THREE.Mesh(new THREE.CapsuleGeometry(0.035, 0.1, 6, 8), bellyMat)
  belly.rotation.x = Math.PI * 0.06; belly.position.set(0, 0.01, 0.02); belly.scale.set(1, 0.6, 1); g.add(belly)
  // Head
  const head = new THREE.Mesh(new THREE.SphereGeometry(0.055, 10, 10), mat)
  head.scale.set(0.9, 0.85, 0.8); head.position.set(0, 0.07, -0.1); g.add(head)
  // Beak - short
  const beak = new THREE.Mesh(new THREE.ConeGeometry(0.015, 0.055, 6), mat)
  beak.rotation.x = -0.2; beak.position.set(0, 0.055, -0.18); g.add(beak)
  // Dorsal fin
  const df = new THREE.Mesh(new THREE.ConeGeometry(0.018, 0.05, 6), mat)
  df.position.set(0, 0.10, -0.01); df.scale.set(0.6, 1, 0.3); g.add(df)
  // Tail flukes
  for (let s = -1; s <= 1; s += 2) {
    const fluke = new THREE.Mesh(new THREE.ConeGeometry(0.016, 0.045, 6), mat)
    fluke.rotation.z = s * 0.5; fluke.position.set(s * 0.04, 0.01, 0.17); g.add(fluke)
  }
  // Eyes - cute
  for (let s = -1; s <= 1; s += 2) {
    const e = new THREE.Mesh(new THREE.SphereGeometry(0.014, 8, 8), new THREE.MeshStandardMaterial({ color: 0x1a1a2e }))
    e.position.set(s * 0.025, 0.075, -0.14); g.add(e)
    const hl = new THREE.Mesh(new THREE.SphereGeometry(0.005, 6, 6), new THREE.MeshStandardMaterial({ color: 0xffffff }))
    hl.position.set(s * 0.028, 0.082, -0.145); g.add(hl)
  }
  // Blush cheeks
  for (let s = -1; s <= 1; s += 2) {
    const blush = new THREE.Mesh(new THREE.SphereGeometry(0.01, 6, 6), new THREE.MeshStandardMaterial({ color: 0xff6b9d, transparent: true, opacity: 0.3 }))
    blush.scale.set(1.2, 0.7, 0.3); blush.position.set(s * 0.035, 0.055, -0.12); g.add(blush)
  }
  // Umbrella accessory
  const umbrellaGroup = new THREE.Group()
  const pole = new THREE.Mesh(new THREE.CylinderGeometry(0.003, 0.003, 0.08, 4), new THREE.MeshStandardMaterial({ color: 0x888899, metalness: 0.3 }))
  pole.position.set(0.07, 0.04, 0.02); umbrellaGroup.add(pole)
  const canopy = new THREE.Mesh(new THREE.SphereGeometry(0.045, 8, 8, 0, Math.PI * 2, 0, Math.PI / 2), umbrellaMat)
  canopy.position.set(0.07, 0.09, 0.02); canopy.rotation.x = Math.PI; umbrellaGroup.add(canopy)
  const rim = new THREE.Mesh(new THREE.TorusGeometry(0.045, 0.004, 6, 12), umbrellaAccent)
  rim.position.set(0.07, 0.05, 0.02); rim.rotation.x = Math.PI / 2; umbrellaGroup.add(rim)
  const tip = new THREE.Mesh(new THREE.SphereGeometry(0.005, 6, 6), new THREE.MeshStandardMaterial({ color: 0xffd700 }))
  tip.position.set(0.07, 0.11, 0.02); umbrellaGroup.add(tip)
  g.add(umbrellaGroup)
  g.userData.umbrella = umbrellaGroup
  // Pectoral fins
  for (let s = -1; s <= 1; s += 2) {
    const pf = new THREE.Mesh(new THREE.ConeGeometry(0.01, 0.035, 4), mat)
    pf.rotation.z = s * 0.5; pf.rotation.x = -0.2
    pf.position.set(s * 0.06, 0.0, -0.04); g.add(pf)
  }
  g.children.forEach(function(ch) { if (ch.isMesh) { ch.userData.isCore = true; ch.userData.isAnimal = true } })
  g.userData.animalType = 'dolphin'
  return g
}

    gl.position.set(s * 0.028, 0.10, -0.09); gl.rotation.x = 0.1; g.add(gl)
    const glLens = new THREE.Mesh(new THREE.CircleGeometry(0.024, 8, 0, Math.PI), glassMat)
    glLens.position.set(s * 0.028, 0.10, -0.085); glLens.rotation.x = -0.2; g.add(glLens)
  }
  // Whiskers
  for (let s = -1; s <= 1; s += 2) {
    for (let w = -1; w <= 1; w += 2) {
      const whisker = new THREE.Mesh(new THREE.CylinderGeometry(0.001, 0.001, 0.035, 4), new THREE.MeshStandardMaterial({ color: 0xcccccc }))
      whisker.position.set(s * 0.035, 0.085 + w * 0.008, -0.13); whisker.rotation.z = s * 0.25; g.add(whisker)
    }
  }
  // Tail
  const tail = new THREE.Mesh(new THREE.CapsuleGeometry(0.01, 0.08, 4, 8), mat)
  tail.rotation.z = 0.8; tail.rotation.x = 0.5; tail.position.set(0, 0.02, 0.10); g.add(tail)
  g.userData.tail = tail
  // Book in arms
  const bookG = new THREE.Group()
  const bookCov = new THREE.Mesh(new THREE.BoxGeometry(0.065, 0.01, 0.045), bookCovMat)
  bookCov.position.set(0.06, -0.02, -0.02); bookG.add(bookCov)
  const bookPag = new THREE.Mesh(new THREE.BoxGeometry(0.055, 0.004, 0.035), bookPagMat)
  bookPag.position.set(0.06, -0.013, -0.02); bookG.add(bookPag)
  const bookLine = new THREE.Mesh(new THREE.BoxGeometry(0.04, 0.002, 0.001), new THREE.MeshStandardMaterial({ color: 0x888888 }))
  bookLine.position.set(0.06, -0.012, -0.02); bookG.add(bookLine)
  g.add(bookG)
  g.userData.book = bookG
  // Legs
  for (let side = -1; side <= 1; side += 2) {
    for (let front = -1; front <= 1; front += 2) {
      const leg = new THREE.Mesh(new THREE.CapsuleGeometry(0.01, 0.02, 4, 6), mat)
      leg.position.set(side * 0.04, -0.04, front * 0.05); g.add(leg)
    }
  }
  g.children.forEach(function(ch) { if (ch.isMesh) { ch.userData.isCore = true; ch.userData.isAnimal = true } })
  g.userData.animalType = 'cat'
  return g
}
