<template>
  <div ref="container" class="iso-office-container">
    <div class="office-toolbar">
      <div class="toolbar-left">
        <span class="toolbar-title">鏁板瓧瀛敓鍔炲叕瀹?/span>
        <span class="toolbar-status" :class="connectionClass">
          <span class="status-dot"></span>
          {{ connectionText }}
        </span>
      </div>
      <div class="toolbar-right">
        <button class="tb-btn" @click="resetCamera" title="閲嶇疆瑙嗚">
          <svg width="14" height="14" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2"><circle cx="12" cy="12" r="3"/><path d="M19.4 15a1.65 1.65 0 0 0 .33 1.82l.06.06a2 2 0 0 1 0 2.83 2 2 0 0 1-2.83 0l-.06-.06a1.65 1.65 0 0 0-1.82-.33 1.65 1.65 0 0 0-1 1.51V21a2 2 0 0 1-2 2 2 2 0 0 1-2-2v-.09A1.65 1.65 0 0 0 9 19.4a1.65 1.65 0 0 0-1.82.33l-.06.06a2 2 0 0 1-2.83 0 2 2 0 0 1 0-2.83l.06-.06A1.65 1.65 0 0 0 4.68 15a1.65 1.65 0 0 0-1.51-1H3a2 2 0 0 1-2-2 2 2 0 0 1 2-2h.09A1.65 1.65 0 0 0 4.6 9a1.65 1.65 0 0 0-.33-1.82l-.06-.06a2 2 0 0 1 0-2.83 2 2 0 0 1 2.83 0l.06.06A1.65 1.65 0 0 0 9 4.68a1.65 1.65 0 0 0 1-1.51V3a2 2 0 0 1 2-2 2 2 0 0 1 2 2v.09a1.65 1.65 0 0 0 1 1.51 1.65 1.65 0 0 0 1.82-.33l.06-.06a2 2 0 0 1 2.83 0 2 2 0 0 1 0 2.83l-.06.06A1.65 1.65 0 0 0 19.4 9a1.65 1.65 0 0 0 1.51 1H21a2 2 0 0 1 2 2 2 2 0 0 1-2 2h-.09a1.65 1.65 0 0 0-1.51 1z"/></svg>
        </button>
        <button class="tb-btn" @click="autoRotate = !autoRotate" :class="{ active: autoRotate }" title="鑷姩鏃嬭浆">
          <svg width="14" height="14" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2"><polyline points="23 4 23 10 17 10"/><path d="M20.49 15a9 9 0 1 1-2.12-9.36L23 10"/></svg>
        </button>
      </div>
    </div>
    <div class="office-legend">
      <div v-for="l in legendItems" :key="l.label" class="legend-item">
        <span class="legend-color" :style="{ background: l.color }"></span>
        <span class="legend-label">{{ l.label }}</span>
      </div>
    </div>
  </div>
</template>
<script setup>
import { ref, onMounted, onBeforeUnmount, nextTick } from 'vue'
import * as THREE from 'three'
import { OrbitControls } from 'three/examples/jsm/controls/OrbitControls.js'
import { CSS2DRenderer, CSS2DObject } from 'three/examples/jsm/renderers/CSS2DRenderer.js'

const props = defineProps({
  agents: { type: Array, default: () => [] },
  collaborations: { type: Array, default: () => [] },
  autoRotate: { type: Boolean, default: true },
})

const container = ref(null)
const connectionClass = ref('connecting')
const connectionText = ref('杩炴帴涓?..')
const autoRotate = ref(props.autoRotate)

const legendItems = [
  { color: '#4ade80', label: '鍦ㄧ嚎 / 宸ヤ綔涓? },
  { color: '#60a5fa', label: '鍙戣█涓? },
  { color: '#fb923c', label: '宸ュ叿璋冪敤' },
  { color: '#f87171', label: '绂荤嚎' },
]

const AGENT_COLORS = {
  orchestrator: '#6c5ce7',
  calculator: '#60a5fa',
  weather: '#4ade80',
  searcher: '#fb923c',
  summarizer: '#f472b6',
}

const STATUS_COLORS = {
  idle: '#4ade80',
  working: '#4ade80',
  speaking: '#60a5fa',
  tool_calling: '#fb923c',
  offline: '#f87171',
}

let scene, camera, renderer, labelRenderer, controls
let animFrameId = null
let clock = new THREE.Clock()
let agentMeshes = {}
let agentLabels = {}
let agentRings = {}
let collaborLines = []
let particleSystem = null

const DESK_POSITIONS = [
  { role: 'orchestrator', x: 0, z: 0, label: 'Orchestrator' },
  { role: 'calculator', x: -3, z: 2.5, label: 'Calculator' },
  { role: 'weather', x: 3, z: 2.5, label: 'Weather' },
  { role: 'searcher', x: -3, z: -2.5, label: 'Searcher' },
  { role: 'summarizer', x: 3, z: -2.5, label: 'Summarizer' },
]

const MEETING_ROOM_POS = { x: 0, z: 5.5 }

function updateAgentStatus(role, status, detail) {
  if (!scene) return
  const mesh = agentMeshes[role]
  if (!mesh) return
  const color = STATUS_COLORS[status] || STATUS_COLORS.idle
  const targetColor = new THREE.Color(color)
  mesh.material.color.lerp(targetColor, 0.3)

  const ring = agentRings[role]
  if (ring) {
    ring.visible = status === 'speaking' || status === 'tool_calling'
    if (ring.visible) { ring.scale.setScalar(1); ring.material.opacity = 0.6 }
  }

  if (status === 'working') mesh.userData.isWorking = true
  else { mesh.userData.isWorking = false; mesh.position.y = 0.25 }

  const lbl = agentLabels[role]
  if (lbl) {
    const statusMap = { idle: '鍦ㄧ嚎', working: '宸ヤ綔涓?..', speaking: '鍙戣█涓?, tool_calling: '璋冪敤宸ュ叿', offline: '绂荤嚎' }
    const s = lbl.element.querySelector('.label-status')
    if (s) { s.textContent = statusMap[status] || status; s.style.color = color }
  }
}

function setConnectionStatus(online) {
  connectionClass.value = online ? 'online' : 'offline'
  connectionText.value = online ? '宸茶繛鎺? : '绂荤嚎'
}

function updateCollaborationLines(links) {
  collaborLines.forEach(l => { scene.remove(l); l.geometry.dispose(); l.material.dispose() })
  collaborLines = []
  if (particleSystem) { scene.remove(particleSystem); particleSystem.geometry.dispose(); particleSystem.material.dispose(); particleSystem = null }
  if (!links || links.length === 0) return
  const positions = []
  links.forEach(link => {
    const fromPos = DESK_POSITIONS.find(p => p.role === link.from)
    const toPos = DESK_POSITIONS.find(p => p.role === link.to)
    if (!fromPos || !toPos) return
    const curve = new THREE.LineCurve3(new THREE.Vector3(fromPos.x, 0.3, fromPos.z), new THREE.Vector3(toPos.x, 0.3, toPos.z))
    const pts = curve.getPoints(20)
    const geometry = new THREE.BufferGeometry().setFromPoints(pts)
    const material = new THREE.LineDashedMaterial({ color: link.color || '#6c5ce7', dashSize: 0.15, gapSize: 0.1, transparent: true, opacity: 0.6 })
    const line = new THREE.Line(geometry, material)
    line.computeLineDistances(); scene.add(line); collaborLines.push(line)
    pts.forEach(p => positions.push(p.x, p.y, p.z))
  })
  if (positions.length > 0) {
    const particleGeo = new THREE.BufferGeometry()
    particleGeo.setAttribute('position', new THREE.Float32BufferAttribute(positions, 3))
    particleSystem = new THREE.Points(particleGeo, new THREE.PointsMaterial({ color: '#8b7cf7', size: 0.08, transparent: true, opacity: 0.7, blending: THREE.AdditiveBlending }))
    scene.add(particleSystem)
  }
}

function resetCamera() {
  if (!controls) return
  controls.target.set(0, 0, 0)
  camera.position.set(8, 7, 8)
  controls.update()
}

function initScene() {
  if (!container.value) return
  const rect = container.value.getBoundingClientRect()
  const w = rect.width || 800; const h = rect.height || 500
  scene = new THREE.Scene(); scene.background = new THREE.Color(0x0d0d1a); scene.fog = new THREE.Fog(0x0d0d1a, 12, 18)
  const aspect = w / h; const vs = 8
  camera = new THREE.OrthographicCamera(-vs * aspect, vs * aspect, vs, -vs, 0.1, 50)
  camera.position.set(8, 7, 8); camera.lookAt(0, 0, 0)
  renderer = new THREE.WebGLRenderer({ antialias: true, alpha: true })
  renderer.setSize(w, h); renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2))
  renderer.shadowMap.enabled = true; renderer.shadowMap.type = THREE.PCFSoftShadowMap
  renderer.toneMapping = THREE.ACESFilmicToneMapping; renderer.toneMappingExposure = 1.2
  container.value.appendChild(renderer.domElement)
  labelRenderer = new CSS2DRenderer()
  labelRenderer.setSize(w, h)
  labelRenderer.domElement.style.position = 'absolute'; labelRenderer.domElement.style.top = '0'; labelRenderer.domElement.style.pointerEvents = 'none'
  container.value.appendChild(labelRenderer.domElement)
  controls = new OrbitControls(camera, renderer.domElement)
  controls.enableDamping = true; controls.dampingFactor = 0.08; controls.target.set(0, 0, 0); controls.minZoom = 3; controls.maxZoom = 20; controls.update()

  scene.add(new THREE.AmbientLight(0x404060, 0.5))
  const dl = new THREE.DirectionalLight(0xffffff, 1.2); dl.position.set(5, 10, 7); dl.castShadow = true; dl.shadow.mapSize.width = 1024; dl.shadow.mapSize.height = 1024; scene.add(dl)
  const fl = new THREE.DirectionalLight(0x6c5ce7, 0.3); fl.position.set(-5, 3, -5); scene.add(fl)
  const rl = new THREE.DirectionalLight(0x60a5fa, 0.4); rl.position.set(0, -3, 8); scene.add(rl)

  createFloor(); createRoomWalls(); createMeetingRoom(); createDesks(); createAgents(); createDecorations()
  animate(); window.addEventListener('resize', onResize)
}

function createFloor() {
  const floor = new THREE.Mesh(new THREE.PlaneGeometry(16, 14), new THREE.MeshStandardMaterial({ color: 0x0d0d1a, roughness: 0.8, metalness: 0.1, side: THREE.DoubleSide }))
  floor.rotation.x = -Math.PI / 2; floor.position.y = -0.05; floor.receiveShadow = true; scene.add(floor)
  const grid = new THREE.GridHelper(14, 14, 0x6c5ce7, 0x2a2a4a)
  grid.material.transparent = true; grid.material.opacity = 0.15; grid.position.y = 0.01; scene.add(grid)
}

function createRoomWalls() {
  const mat = new THREE.MeshStandardMaterial({ color: 0x141428, transparent: true, opacity: 0.3, side: THREE.DoubleSide, roughness: 0.6, metalness: 0.3 })
  const h = 2.5; [[0, h/2, -7, 16, h, 0.1],[0, h/2, 7, 16, h, 0.1],[-8, h/2, 0, 0.1, h, 14],[8, h/2, 0, 0.1, h, 14]].forEach(w => {
    const m = new THREE.Mesh(new THREE.BoxGeometry(w[3], w[4], w[5]), mat); m.position.set(w[0], w[1], w[2]); scene.add(m)
  })
}

function createMeetingRoom() {
  const { x, z } = MEETING_ROOM_POS
  const gm = new THREE.MeshPhysicalMaterial({ color: 0x6c5ce7, transparent: true, opacity: 0.08, roughness: 0.1, side: THREE.DoubleSide })
  [[x-1.2, 1.8/2, z, 0.05, 1.8, 2.4],[x+1.2, 1.8/2, z, 0.05, 1.8, 2.4],[x, 1.8/2, z-1.2, 2.4, 1.8, 0.05]].forEach(p => {
    const m = new THREE.Mesh(new THREE.BoxGeometry(p[3], p[4], p[5]), gm); m.position.set(p[0], p[1], p[2]); scene.add(m)
  })
  const floor = new THREE.Mesh(new THREE.PlaneGeometry(2.2, 2.2), new THREE.MeshStandardMaterial({ color: 0x1a1a35, roughness: 0.3, metalness: 0.6, transparent: true, opacity: 0.5, side: THREE.DoubleSide }))
  floor.rotation.x = -Math.PI / 2; floor.position.set(x, 0.01, z); scene.add(floor)
  const table = new THREE.Mesh(new THREE.BoxGeometry(1.6, 0.08, 1), new THREE.MeshStandardMaterial({ color: 0x2a2a50, roughness: 0.4, metalness: 0.2 }))
  table.position.set(x, 0.2, z); table.castShadow = true; scene.add(table)
  const div = document.createElement('div'); div.className = 'label-3d label-meeting'; div.innerHTML = '<span>浼氳瀹?/span>'
  const lbl = new CSS2DObject(div); lbl.position.set(x, 2.2, z); scene.add(lbl)
}

function createDesks() {
  DESK_POSITIONS.forEach(({ role, x, z }) => {
    const g = new THREE.Group(); g.position.set(x, 0, z)
    const base = new THREE.Mesh(new THREE.BoxGeometry(1.2, 0.3, 0.8), new THREE.MeshStandardMaterial({ color: 0x1a1a30, roughness: 0.5, metalness: 0.3 }))
    base.position.y = 0.15; base.castShadow = true; base.receiveShadow = true; g.add(base)
    const glow = new THREE.Mesh(new THREE.BoxGeometry(1.25, 0.02, 0.85), new THREE.MeshBasicMaterial({ color: 0x6c5ce7, transparent: true, opacity: 0.05 }))
    glow.position.y = 0.31; g.add(glow)
    const fm = new THREE.MeshStandardMaterial({ color: 0x3a3a5a, metalness: 0.5, roughness: 0.3 })
    ;[[-0.5, 0.05, -0.35],[0.5, 0.05, -0.35],[-0.5, 0.05, 0.35],[0.5, 0.05, 0.35]].forEach(fp => {
      const foot = new THREE.Mesh(new THREE.CylinderGeometry(0.04, 0.05, 0.1), fm); foot.position.set(fp[0], fp[1], fp[2]); g.add(foot)
    })
    scene.add(g)
  })
}

function createAgents() {
  DESK_POSITIONS.forEach(({ role, x, z, label }) => {
    const color = new THREE.Color(AGENT_COLORS[role] || '#6c5ce7')
    const g = new THREE.Group(); g.position.set(x, 0.6, z); scene.add(g)
    const body = new THREE.Mesh(new THREE.BoxGeometry(0.5, 0.5, 0.5), new THREE.MeshPhysicalMaterial({ color, roughness: 0.2, metalness: 0.6, clearcoat: 0.3, clearcoatRoughness: 0.4, emissive: color, emissiveIntensity: 0.15 }))
    body.castShadow = true; body.position.y = 0.25; body.userData.role = role; g.add(body)
    const head = new THREE.Mesh(new THREE.SphereGeometry(0.18, 16, 16), new THREE.MeshPhysicalMaterial({ color, roughness: 0.15, metalness: 0.5, emissive: color, emissiveIntensity: 0.2 }))
    head.position.y = 0.55; head.castShadow = true; g.add(head)
    const am = new THREE.MeshBasicMaterial({ color: 0x4ade80 })
    const ant = new THREE.Mesh(new THREE.CylinderGeometry(0.02, 0.02, 0.15), am); ant.position.y = 0.7; g.add(ant)
    const tip = new THREE.Mesh(new THREE.SphereGeometry(0.035, 8, 8), am); tip.position.y = 0.78; g.add(tip)
    const ring = new THREE.Mesh(new THREE.RingGeometry(0.3, 0.45, 32), new THREE.MeshBasicMaterial({ color, transparent: true, opacity: 0, side: THREE.DoubleSide, depthWrite: false }))
    ring.rotation.x = -Math.PI / 2; ring.position.y = 0.02; ring.visible = false; g.add(ring)
    agentMeshes[role] = body; agentRings[role] = ring
    const gf = new THREE.Mesh(new THREE.CircleGeometry(0.6, 24), new THREE.MeshBasicMaterial({ color, transparent: true, opacity: 0.08 }))
    gf.rotation.x = -Math.PI / 2; gf.position.y = 0.01; g.add(gf)
    const div = document.createElement('div'); div.className = 'label-3d'
    const rm = { orchestrator: 'Orchestrator', calculator: 'Calculator', weather: 'Weather', searcher: 'Searcher', summarizer: 'Summarizer' }
    div.innerHTML = '<div class="label-name">' + (rm[role] || label) + '</div><div class="label-status" style="color:#4ade80">鍦ㄧ嚎</div>'
    const lbl = new CSS2DObject(div); lbl.position.set(x, 1.6, z); scene.add(lbl); agentLabels[role] = lbl
  })
}

function createDecorations() {
  const pm = new THREE.MeshStandardMaterial({ color: 0x1a5c3a, roughness: 0.8 })
  const pt = new THREE.MeshStandardMaterial({ color: 0x3a3a5a, roughness: 0.6, metalness: 0.2 })
  ;[[-6, -4],[6, -4],[-6, 4],[6, 4]].forEach(([px, pz]) => {
    const g = new THREE.Group(); g.position.set(px, 0, pz)
    const pot = new THREE.Mesh(new THREE.CylinderGeometry(0.2, 0.15, 0.2), pt); pot.position.y = 0.1; g.add(pot)
    for (let i = 0; i < 5; i++) {
      const leaf = new THREE.Mesh(new THREE.SphereGeometry(0.08 + Math.random() * 0.08, 6, 6), pm)
      const a = (i / 5) * Math.PI * 2; const r = 0.1 + Math.random() * 0.08
      leaf.position.set(Math.cos(a) * r, 0.25 + Math.random() * 0.1, Math.sin(a) * r); g.add(leaf)
    }
    scene.add(g)
  })
  const pc = 150; const pg = new THREE.BufferGeometry(); const pp = new Float32Array(pc * 3)
  for (let i = 0; i < pc * 3; i++) { pp[i] = (Math.random() - 0.5) * 16; if (i % 3 === 1) pp[i] = Math.random() * 3 }
  pg.setAttribute('position', new THREE.BufferAttribute(pp, 3))
  scene.add(new THREE.Points(pg, new THREE.PointsMaterial({ color: 0x6c5ce7, size: 0.03, transparent: true, opacity: 0.3, blending: THREE.AdditiveBlending, depthWrite: false })))
}

function animate() {
  animFrameId = requestAnimationFrame(animate); const t = clock.getElapsedTime()
  Object.values(agentMeshes).forEach(m => { if (m.userData.isWorking) m.position.y = 0.25 + Math.sin(t * 3 + (m.userData.role || '').length) * 0.03 })
  Object.entries(agentRings).forEach(([, r]) => { if (r.visible) { const s = 1 + Math.sin(t * 2) * 0.3; r.scale.setScalar(s); r.material.opacity = 0.3 + Math.sin(t * 2) * 0.2 } })
  collaborLines.forEach(l => { l.material.dashOffset -= 0.03 })
  if (particleSystem) { const pos = particleSystem.geometry.attributes.position.array; for (let i = 1; i < pos.length; i += 3) pos[i] += Math.sin(t * 0.5 + i) * 0.001; particleSystem.geometry.attributes.position.needsUpdate = true }
  if (autoRotate.value && controls) { const r = Math.sqrt(camera.position.x**2 + camera.position.z**2); const a = Math.atan2(camera.position.z, camera.position.x) + 0.0004; camera.position.x = r * Math.cos(a); camera.position.z = r * Math.sin(a); camera.lookAt(controls.target) }
  controls.update(); renderer.render(scene, camera); labelRenderer.render(scene, camera)
}

function onResize() {
  if (!container.value || !camera || !renderer || !labelRenderer) return
  const rect = container.value.getBoundingClientRect(); const w = rect.width || 800; const h = rect.height || 500; const aspect = w / h; const vs = 8
  camera.left = -vs * aspect; camera.right = vs * aspect; camera.top = vs; camera.bottom = -vs; camera.updateProjectionMatrix()
  renderer.setSize(w, h); labelRenderer.setSize(w, h)
}

onMounted(() => { nextTick(() => initScene()) })
onBeforeUnmount(() => { if (animFrameId) cancelAnimationFrame(animFrameId); window.removeEventListener('resize', onResize); renderer?.dispose(); renderer?.domElement?.remove(); labelRenderer?.domElement?.remove() })

defineExpose({ updateAgentStatus, setConnectionStatus, updateCollaborationLines, resetCamera })
</script>
